00619        of the str field (the human readable name) */
00620     swig_module_info *iter = start;
00621     do {
00622       register size_t i = 0;
00623       for (; i < iter->size; ++i) {
00624         if (iter->types[i]->str && (SWIG_TypeEquiv(iter->types[i]->str, name)))
00625           return iter->types[i];
00626       }
00627       iter = iter->next;
00628     } while (iter != end);
00629   }
00630   
00631   /* neither found a match */
00632   return 0;
00633 }

00635 /* 
00636    Pack binary data into a string
00637 */
00638 SWIGRUNTIME char *
00639 SWIG_PackData(char *c, void *ptr, size_t sz) {
00640   static const char hex[17] = "0123456789abcdef";
00641   register const unsigned char *u = (unsigned char *) ptr;
00642   register const unsigned char *eu =  u + sz;
00643   for (; u != eu; ++u) {
00644     register unsigned char uu = *u;
00645     *(c++) = hex[(uu & 0xf0) >> 4];
00646     *(c++) = hex[uu & 0xf];
00647   }
00648   return c;
00649 }

00651 /* 
00652    Unpack binary data from a string
00653 */
00654 SWIGRUNTIME const char *
00655 SWIG_UnpackData(const char *c, void *ptr, size_t sz) {
00656   register unsigned char *u = (unsigned char *) ptr;
00657   register const unsigned char *eu = u + sz;
00658   for (; u != eu; ++u) {
00659     register char d = *(c++);
00660     register unsigned char uu;
00661     if ((d >= '0') && (d <= '9'))
00662       uu = ((d - '0') << 4);
00663     else if ((d >= 'a') && (d <= 'f'))
00664       uu = ((d - ('a'-10)) << 4);
00665     else 
00666       return (char *) 0;
00667     d = *(c++);
00668     if ((d >= '0') && (d <= '9'))
00669       uu |= (d - '0');
00670     else if ((d >= 'a') && (d <= 'f'))
00671       uu |= (d - ('a'-10));
00672     else 
00673       return (char *) 0;
00674     *u = uu;
00675   }
00676   return c;
00677 }

01060 /* Convert a pointer value */
01061 SWIGRUNTIME int
01062 SWIG_Tcl_ConvertPtrFromString(Tcl_Interp *interp, const char *c, void **ptr, swig_type_info *ty, int flags) {
01063   swig_cast_info *tc;
01064   /* Pointer values must start with leading underscore */
01065   while (*c != '_') {
01066     *ptr = (void *) 0;
01067     if (strcmp(c,"NULL") == 0) return SWIG_OK;
01068 
01069     /* Empty string: not a pointer */
01070     if (*c == 0) return SWIG_ERROR; 
01071 
01072     /* Hmmm. It could be an object name. */
01073 
01074     /* Check if this is a command at all. Prevents <c> cget -this         */
01075     /* from being called when c is not a command, firing the unknown proc */
01076     if (Tcl_VarEval(interp,"info commands ", c, (char *) NULL) == TCL_OK) {
01077       Tcl_Obj *result = Tcl_GetObjResult(interp);
01078       if (*(Tcl_GetStringFromObj(result, NULL)) == 0) {
01079         /* It's not a command, so it can't be a pointer */
01080         Tcl_ResetResult(interp);
01081         return SWIG_ERROR;
01082       }
01083     } else {
01084       /* This will only fail if the argument is multiple words. */
01085       /* Multiple words are also not commands.                  */
01086       Tcl_ResetResult(interp);
01087       return SWIG_ERROR;
01088     }
01089 
01090     /* Check if this is really a SWIG pointer */
01091     if (Tcl_VarEval(interp,c," cget -this", (char *) NULL) != TCL_OK) {
01092       Tcl_ResetResult(interp);
01093       return SWIG_ERROR;
01094     }
01095 
01096     c = Tcl_GetStringFromObj(Tcl_GetObjResult(interp), NULL);
01097   }
01098 
01099   c++;
01100   c = SWIG_UnpackData(c,ptr,sizeof(void *));
01101   if (ty) {
01102     tc = c ? SWIG_TypeCheck(c,ty) : 0;
01103     if (!tc) {
01104       return SWIG_ERROR;
01105     }
01106     if (flags & SWIG_POINTER_DISOWN) {
01107       SWIG_Disown((void *) *ptr);
01108     }
01109     {
01110       int newmemory = 0;
01111       *ptr = SWIG_TypeCast(tc,(void *) *ptr,&newmemory);
01112       assert(!newmemory); /* newmemory handling not yet implemented */
01113     }
01114   }
01115   return SWIG_OK;
01116 }

02118 /* -----------------------------------------------------------------------------
02119  * Type initialization:
02120  * This problem is tough by the requirement that no dynamic 
02121  * memory is used. Also, since swig_type_info structures store pointers to 
02122  * swig_cast_info structures and swig_cast_info structures store pointers back
02123  * to swig_type_info structures, we need some lookup code at initialization. 
02124  * The idea is that swig generates all the structures that are needed. 
02125  * The runtime then collects these partially filled structures. 
02126  * The SWIG_InitializeModule function takes these initial arrays out of 
02127  * swig_module, and does all the lookup, filling in the swig_module.types
02128  * array with the correct data and linking the correct swig_cast_info
02129  * structures together.
02130  *
02131  * The generated swig_type_info structures are assigned staticly to an initial 
02132  * array. We just loop through that array, and handle each type individually.
02133  * First we lookup if this type has been already loaded, and if so, use the
02134  * loaded structure instead of the generated one. Then we have to fill in the
02135  * cast linked list. The cast data is initially stored in something like a
02136  * two-dimensional array. Each row corresponds to a type (there are the same
02137  * number of rows as there are in the swig_type_initial array). Each entry in
02138  * a column is one of the swig_cast_info structures for that type.
02139  * The cast_initial array is actually an array of arrays, because each row has
02140  * a variable number of columns. So to actually build the cast linked list,
02141  * we find the array of casts associated with the type, and loop through it 
02142  * adding the casts to the list. The one last trick we need to do is making
02143  * sure the type pointer in the swig_cast_info struct is correct.
02144  *
02145  * First off, we lookup the cast->type name to see if it is already loaded. 
02146  * There are three cases to handle:
02147  *  1) If the cast->type has already been loaded AND the type we are adding
02148  *     casting info to has not been loaded (it is in this module), THEN we
02149  *     replace the cast->type pointer with the type pointer that has already
02150  *     been loaded.
02151  *  2) If BOTH types (the one we are adding casting info to, and the 
02152  *     cast->type) are loaded, THEN the cast info has already been loaded by
02153  *     the previous module so we just ignore it.
02154  *  3) Finally, if cast->type has not already been loaded, then we add that
02155  *     swig_cast_info to the linked list (because the cast->type) pointer will
02156  *     be correct.
02157  * ----------------------------------------------------------------------------- */
