263 proc updatevalidtopics { s } {
264 global SAL_DIR SAL_SUBSYSTEMS SAL_SID SAL_WORK_DIR
265  stdlog "Creating svcSAL.cpp.[set s]_topics"
266  stdlog "Creating svcSAL_[set s]_topics.h"
267  set fo3 [open $SAL_DIR/code/simd/svcSAL.cpp.[set s]_topics w]
268  set fo2 [open $SAL_DIR/code/include/sal/svcSAL_[set s]_topics.h w]
269  puts $fo2 "#if !defined(__INCLUDED_SAL_[set s]_TOPICS)"
270  puts $fo2 "#define __INCLUDED_SAL_[set s]_TOPICS"
271  set topicdirs [lsort [glob $SAL_WORK_DIR/ospl-$s*]]
272  foreach t $topicdirs {
273  set id [string range [file tail [file root $t]] 5 end]
274  puts $fo2 "#include \"svcSAL_[set id]_iid.h\""
275  puts $fo2 "#define SAL_SHMEM_KEY_[set id] 0x[calcshmid $id]"
276  puts $fo3 "
277  case SAL_IID_$id :
278  salTopic = ([set id] *)new [set id]();
279  result = SAL__OK;
280  break;
281 "
282  }
283  foreach t $topicdirs {
284  set id [string range [file tail [file root $t]] 5 end]
285  set mdid [lindex [exec md5sum $t/$id.idl] 0]
286  set tid "0x[string range $mdid 26 end]"
287  set SAL_SID($id) $tid
288  puts $fo2 "#define SAL_SID_$id $tid"
289  puts stdout "Done $id"
290  }
291  puts $fo2 "#endif\n"

293  close $fo3
